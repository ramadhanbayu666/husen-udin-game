<script type="module">

import * as THREE from 'three';



// ===== 1. SETUP SCENE =====

const scene = new THREE.Scene();

scene.fog = new THREE.Fog(0x87CEEB, 30, 100);

const camera = new THREE.PerspectiveCamera(70, innerWidth/innerHeight, 0.1, 200);

const renderer = new THREE.WebGLRenderer({canvas: document.getElementById('c'), antialias: true});

renderer.setSize(innerWidth, innerHeight);

renderer.setClearColor(0x87CEEB);

renderer.shadowMap.enabled = true;



// ===== 2. LIGHTING =====

scene.add(new THREE.AmbientLight(0xffffff, 0.7));

const sun = new THREE.DirectionalLight(0xffffff, 1);

sun.position.set(10, 20, 10);

sun.castShadow = true;

scene.add(sun);



// ===== 3. GARIS START "BAYU BUNDAR" =====

const canvas = document.createElement('canvas');

canvas.width = 512; canvas.height = 128;

const ctx = canvas.getContext('2d');

ctx.fillStyle = '#ffffff';

ctx.fillRect(0, 0, 512, 128);

ctx.font = 'bold 48px Arial';

ctx.fillStyle = '#ff0000';

ctx.textAlign = 'center';

ctx.fillText('BAYU BUNDAR', 256, 50);

ctx.fillStyle = '#000000';

ctx.fillText('START', 256, 100);

const startTex = new THREE.CanvasTexture(canvas);

const startLine = new THREE.Mesh(

new THREE.PlaneGeometry(12, 3, 24, 24),

new THREE.MeshBasicMaterial({map: startTex})

);

startLine.rotation.x = -Math.PI/2;

startLine.position.set(0, 0.01, 0);

scene.add(startLine);



// ===== 4. JALAN & RUMPUT - 24 SEGMENT =====

const road = new THREE.Mesh(

new THREE.PlaneGeometry(12, 500, 24, 24),

new THREE.MeshStandardMaterial({color: 0x8B4513, roughness: 0.9})

);

road.rotation.x = -Math.PI/2;

road.receiveShadow = true;

scene.add(road);



const grassL = new THREE.Mesh(

new THREE.PlaneGeometry(20, 500, 24, 24),

new THREE.MeshStandardMaterial({color: 0x228B22})

);

grassL.rotation.x = -Math.PI/2;

grassL.position.set(-16, -0.01, 0);

scene.add(grassL);



const grassR = grassL.clone();

grassR.position.x = 16;

scene.add(grassR);



// ===== 5. FUNGSI BIKIN KARAKTER GEOMETRY 24 SEGMENT =====

function buatKarakter(warnaBaju, nama) {

const char = new THREE.Group();



// Kepala botak - SphereGeometry 24 segment

const kepala = new THREE.Mesh(

new THREE.SphereGeometry(0.5, 24, 24),

new THREE.MeshStandardMaterial({color: 0xFFDBAC, roughness: 0.5})

);

kepala.position.y = 1.3;

kepala.castShadow = true;

char.add(kepala);



// Mata - SphereGeometry 24 segment

const mata1 = new THREE.Mesh(new THREE.SphereGeometry(0.08, 24, 24), new THREE.MeshBasicMaterial({color: 0x000000}));

mata1.position.set(-0.15, 1.35, 0.45);

char.add(mata1);

const mata2 = mata1.clone();

mata2.position.x = 0.15;

char.add(mata2);



// Badan - CylinderGeometry 24 segment

const badan = new THREE.Mesh(

new THREE.CylinderGeometry(0.4, 0.5, 0.8, 24),

new THREE.MeshStandardMaterial({color: warnaBaju, metalness: 0.1, roughness: 0.7})

);

badan.position.y = 0.6;

badan.castShadow = true;

char.add(badan);



// Huruf di baju

const hurufCanvas = document.createElement('canvas');

hurufCanvas.width = 128; hurufCanvas.height = 128;

const hCtx = hurufCanvas.getContext('2d');

hCtx.fillStyle = '#ffffff';

hCtx.font = 'bold 100px Arial';

hCtx.textAlign = 'center';

hCtx.fillText(nama, 64, 90);

const hurufTex = new THREE.CanvasTexture(hurufCanvas);

const huruf = new THREE.Mesh(

new THREE.PlaneGeometry(0.3, 0.3),

new THREE.MeshBasicMaterial({map: hurufTex, transparent: true})

);

huruf.position.set(0, 0.6, 0.51);

char.add(huruf);



// Tangan - CylinderGeometry 24 segment

const tanganKiri = new THREE.Mesh(

new THREE.CylinderGeometry(0.1, 0.1, 0.6, 24),

new THREE.MeshStandardMaterial({color: 0xFFDBAC})

);

tanganKiri.position.set(-0.6, 0.6, 0);

tanganKiri.rotation.z = Math.PI/4;

tanganKiri.castShadow = true;

char.add(tanganKiri);

const tanganKanan = tanganKiri.clone();

tanganKanan.position.x = 0.6;

tanganKanan.rotation.z = -Math.PI/4;

char.add(tanganKanan);



// Kaki - CylinderGeometry 24 segment

const kakiKiri = new THREE.Mesh(

new THREE.CylinderGeometry(0.12, 0.12, 0.5, 24),

new THREE.MeshStandardMaterial({color: 0x0000ff})

);

kakiKiri.position.set(-0.2, 0.25, 0);

kakiKiri.castShadow = true;

char.add(kakiKiri);

const kakiKanan = kakiKiri.clone();

kakiKanan.position.x = 0.2;

char.add(kakiKanan);



return char;

}



// ===== 6. PLAYER: HUSEN =====

const player = buatKarakter(0xffff00, 'H'); // Kuning

player.position.set(0, 0, 2);

scene.add(player);



// ===== 7. OBSTACLE PETUALANGAN - 24 SEGMENT =====

const obstacles = [];

function spawnObstacle(z) {

const type = Math.floor(Math.random()*3);

let obs;

if(type === 0){ // Batu - DodecahedronGeometry 24 segment

obs = new THREE.Mesh(

new THREE.DodecahedronGeometry(0.6, 0),

new THREE.MeshStandardMaterial({color: 0x808080, roughness: 1})

);

} else if(type === 1){ // Pohon - ConeGeometry 24 segment

obs = new THREE.Group();

const batang = new THREE.Mesh(new THREE.CylinderGeometry(0.2, 0.2, 1.5, 24), new THREE.MeshStandardMaterial({color: 0x8B4513}));

const daun = new THREE.Mesh(new THREE.ConeGeometry(0.8, 2, 24), new THREE.MeshStandardMaterial({color: 0x006400}));

daun.position.y = 1.5;

obs.add(batang, daun);

} else { // Durian - SphereGeometry 24 segment

obs = new THREE.Mesh(

new THREE.SphereGeometry(0.5, 24, 24),

new THREE.MeshStandardMaterial({color: 0x2d5016, roughness: 1})

);

}

obs.position.set((Math.random()-0.5)*8, type===1?0:0.5, z);

obs.castShadow = true;

scene.add(obs);

obstacles.push(obs);

}

for(let i = 0; i < 20; i++) spawnObstacle(-20 - i*10);



// ===== 8. KONTROL PC + ANDROID =====

let moveX = 0, isJump = false, velocityY = 0, onGround = true;

const keys = {};

addEventListener('keydown', e => {

keys[e.key.toLowerCase()] = true;

if(e.key === ' ' || e.key === 'w' || e.key === 'ArrowUp') jump();

});

addEventListener('keyup', e => keys[e.key.toLowerCase()] = false);



// Touch Android

document.getElementById('btnLeft').ontouchstart = () => moveX = -1;

document.getElementById('btnLeft').ontouchend = () => moveX = 0;

document.getElementById('btnRight').ontouchstart = () => moveX = 1;

document.getElementById('btnRight').ontouchend = () => moveX = 0;

document.getElementById('btnJump').ontouchstart = () => jump();



function jump() {

if(onGround) {

velocityY = 0.35;

onGround = false;

}

}



// ===== 9. GAME LOOP LOMBA LARI =====

let score = 0, lives = 3, gameSpeed = 0.25, gameStarted = false;

camera.position.set(0, 4, 10);



// Tampilan START

document.getElementById('startText').style.display = 'block';

setTimeout(() => {

document.getElementById('startText').style.display = 'none';

gameStarted = true;

}, 2000);



function animate() {

requestAnimationFrame(animate);

if(!gameStarted) return;



// Kontrol PC

if(keys['a'] || keys['arrowleft']) moveX = -1;

else if(keys['d'] || keys['arrowright']) moveX = 1;

else if(!document.querySelector('.btn:active')) moveX = 0;



// Gerak player

player.position.x += moveX * 0.15;

player.position.x = Math.max(-5, Math.min(5, player.position.x));



// Lompat + Gravitasi

player.position.y += velocityY;

velocityY -= 0.02;

if(player.position.y <= 0) {

player.position.y = 0;

velocityY = 0;

onGround = true;

}



// Animasi lari

const runCycle = Date.now() * 0.01;

player.rotation.z = Math.sin(runCycle) * 0.1;

player.children[5].rotation.x = Math.sin(runCycle) * 0.5; // kaki kiri

player.children[6].rotation.x = -Math.sin(runCycle) * 0.5; // kaki kanan



// Jalan maju terus

player.position.z -= gameSpeed;

score = Math.floor(-player.position.z);

document.getElementById('score').textContent = score;



// Kamera ngikutin

camera.position.set(player.position.x*0.5, 4, player.position.z + 10);

camera.lookAt(player.position.x, 1, player.position.z - 5);



// Cek tabrakan obstacle

obstacles.forEach((obs, i) => {

if(obs.position.distanceTo(player.position) < 1.2) {

scene.remove(obs);

obstacles.splice(i, 1);

lives--;

document.getElementById('lives').textContent = lives;

spawnObstacle(player.position.z - 100);

if(lives <= 0) {

alert(`GAME OVER\nJarak: ${score}m\n\nGame by Bayu Bundar`);

location.reload();

}

}

// Respawn obstacle

if(obs.position.z > player.position.z + 20) {

obs.position.z = player.position.z - 100;

obs.position.x = (Math.random()-0.5)*8;

}

});



gameSpeed += 0.0001; // Makin cepet

renderer.render(scene, camera);

}

animate();



// Resize

addEventListener('resize', () => {

camera.aspect = innerWidth/innerHeight;

camera.updateProjectionMatrix();

renderer.setSize(innerWidth, innerHeight);

});
